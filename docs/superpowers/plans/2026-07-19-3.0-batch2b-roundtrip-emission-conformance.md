### Task 7: PROV-XML round-trips empty-string values and non-NCName attribute names (#224, #289)

**Goal:** An attribute whose value is `""` survives the PROV-XML round trip instead of being dropped (#224), and attribute *names* containing NCName-illegal characters serialize without `ValueError` and round-trip losslessly via reversible escaping (#289) — letting `strategies.py` drop both the `min_size=1` value restriction and the batch-2a `local_part`/`attr_key_part` split.

```python
_NCNAME_START_CHARS = (
    "A-Za-z_\xc0-\xd6\xd8-\xf6\xf8-\u02ff\u0370-\u037d\u037f-\u1fff"
    "\u200c-\u200d\u2070-\u218f\u2c00-\u2fef\u3001-\ud7ff\uf900-\ufdcf"
    "\ufdf0-\ufffd\U00010000-\U000effff"
)
_NCNAME_CHARS = _NCNAME_START_CHARS + "\\-.0-9\xb7\u0300-\u036f\u203f-\u2040"
_NCNAME_START_RE = re.compile(f"[{_NCNAME_START_CHARS}]")
_NCNAME_CHAR_RE = re.compile(f"[{_NCNAME_CHARS}]")
# One escaped char: _x + 4 (BMP) or 8 (astral) uppercase hex digits + _
_XML_NAME_ESCAPE_RE = re.compile(r"_x[0-9A-F]{4}(?:[0-9A-F]{4})?_")


def _escape_ncname_localpart(local: str) -> str:
    """Escape NCName-illegal characters in an attribute-name local part.

    Uses the ``_xHHHH_`` convention; a literal run that itself looks like
    an escape sequence gets its introducing underscore self-escaped as
    ``_x005F_`` so decoding is always the exact inverse (#289).
    """
    out = []
    for i, char in enumerate(local):
        char_re = _NCNAME_START_RE if i == 0 else _NCNAME_CHAR_RE
        if char == "_" and _XML_NAME_ESCAPE_RE.match(local, i):
            out.append("_x005F_")
        elif char_re.fullmatch(char):
            out.append(char)
        else:
            code = ord(char)
            out.append(f"_x{code:04X}_" if code <= 0xFFFF else f"_x{code:08X}_")
    return "".join(out)


def _unescape_ncname_localpart(local: str) -> str:
    return _XML_NAME_ESCAPE_RE.sub(
        lambda m: chr(int(m.group(0)[2:-1], 16)), local
    )
```
